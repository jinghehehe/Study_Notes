class Solution {
public:
    void backTrack(vector<int>& nums, vector<int>& st){
        if(st.size() == nums.size()){
            res.push_back(st);
            return;
        }
        else{
            for(int i=0; i<nums.size(); i++){
                if(find(st.begin(),st.end(),nums[i])!=st.end()){//未找到
                    continue;
                }                    
                else{
                    st.push_back(nums[i]);
                    backTrack(nums, st);
                    st.pop_back();
                }
            }
        }
            
    }
    vector<vector<int>> permute(vector<int>& nums) {
        int len = nums.size();
        if(len == 0) return res;
        backTrack(nums, st);
        return res;
    }
private:
    vector<vector<int> > res;
    vector<int> st;
};
